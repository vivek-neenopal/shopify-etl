-- sql/fact_current_inventory.sql
-- Logic: 
-- 1. "Retail First": Prioritize Retail data over Wholesale.
-- 2. "Latest Variant": Use DISTINCT ON to pick the NEWEST created_at per SKU.
-- 3. No Aggregation: Uses raw values from the latest variant.

CREATE TEMP TABLE tmp_fact_current_inventory ON COMMIT DROP AS
WITH retail_data AS (
    -- 1. RETAIL: Deduplicate by picking the LATEST created_at per SKU
    SELECT DISTINCT ON (UPPER(TRIM(v.sku)))
        UPPER(TRIM(v.sku)) as sku,
        i.available,
        i.committed,
        i.on_hand,
        i.incoming,
        i.reserved,
        i.variant_id,
        i.product_id,
        'retail' AS source_system
    FROM staging_retail_fact_current_inventory i
    JOIN staging_retail_dim_product_variants v ON i.variant_id = v.variant_id
    JOIN staging_retail_dim_products p ON i.product_id = p.product_id
    WHERE v.sku IS NOT NULL AND TRIM(v.sku) <> '' and p.status = 'ACTIVE'
    -- Sort by SKU (required for DISTINCT ON) and then by Date DESC (Newest First)
    ORDER BY UPPER(TRIM(v.sku)), v.created_at DESC
),
wholesale_data AS (
    -- 2. WHOLESALE: Deduplicate by picking the LATEST created_at per SKU
    SELECT DISTINCT ON (UPPER(TRIM(v.sku)))
        UPPER(TRIM(v.sku)) as sku,
        i.available,
        i.committed,
        i.on_hand,
        i.incoming,
        i.reserved,
        i.variant_id,
        i.product_id,
        'wholesale' AS source_system
    FROM staging_wholesale_fact_current_inventory i
    JOIN staging_wholesale_dim_product_variants v ON i.variant_id = v.variant_id
    JOIN staging_wholesale_dim_products p ON i.product_id = p.product_id
    WHERE v.sku IS NOT NULL AND TRIM(v.sku) <> '' and p.status = 'ACTIVE'
    ORDER BY UPPER(TRIM(v.sku)), v.created_at DESC
),
final_merged_inventory AS (
    -- A. Take ALL Retail Data (Primary Source)
    SELECT * FROM retail_data

    UNION ALL

    -- B. Take Wholesale Data ONLY if SKU does NOT exist in Retail (Gap Fill)
    SELECT * FROM wholesale_data w
    WHERE w.sku NOT IN (SELECT sku FROM retail_data)
)
SELECT 
    sku,
    available,
    committed,
    on_hand,
    incoming,
    reserved,
    variant_id,
    product_id,
    source_system
FROM final_merged_inventory;

        t.available,

        t.source_system,

    ON CONFLICT (sku)
